224 static PRESSURE_StatusTypeDef LPS25HB_I2C_ReadRawPressure(int32_t *raw_press)
225 {
226  uint8_t buffer[3], i;
227  uint32_t tempVal=0;
228 
229  /* Read the register content */
230 
231  if(LPS25HB_IO_Read(buffer, LPS25HB_SlaveAddress, (LPS25HB_PRESS_POUT_XL_ADDR | LPS25HB_I2C_MULTIPLEBYTE_CMD), 3) != PRESSURE_OK)
232  {
233  return PRESSURE_ERROR;
234  }
235 
236  /* Build the raw data */
237  for (i = 0 ; i < 3 ; i++)
238  tempVal |= (((uint32_t) buffer[i]) << (8 * i));
239 
240  /* convert the 2's complement 24 bit to 2's complement 32 bit */
241  if (tempVal & 0x00800000)
242  tempVal |= 0xFF000000;
243 
244  /* return the built value */
245  *raw_press = ((int32_t) tempVal);
246 
247  return PRESSURE_OK;
248 }
