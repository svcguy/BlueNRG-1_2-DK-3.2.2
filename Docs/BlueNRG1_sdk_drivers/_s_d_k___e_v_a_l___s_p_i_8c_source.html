576 void IrqSpiHandlerCallback(void)
577 {
578  /* Check if SPI RX interrupt event occured */
579  if(SPI_GetITStatus(SPI_IT_RX) == SET) {
580 
581  if(buffer_idx==0) { /* First reception is discarded, dummy data from sensor */
582  SPI_ReceiveData();
583  }
584  else {
585  spi_buffer_ptr[buffer_idx-1] = SPI_ReceiveData();
586  }
587 
588  /* When all the bytes are received, the flag is set */
589  if(byte_to_send_max == buffer_idx)
590  spi_eot = SET;
591 
592  /* Increase the buffer index */
593  buffer_idx++;
594  }
595 
596  /* Check if SPI RX interrupt event occured */
597  if(SPI_GetITStatus(SPI_IT_TX) == SET) {
598 
599  if(byte_to_send>0) {
600  if(spi_irq_state == SEND_READ_REG) {
601  spi_irq_state = SEND_START;
602 
603  /* Send the header byte for read operation from register_addr */
604  SPI_SendData(0x80 | register_addr);
605  }
606  else if(spi_irq_state == SEND_WRITE_REG) {
607  spi_irq_state = SEND_WRITE_DATA;
608 
609  /* Send the header byte for write operation from register_addr */
610  SPI_SendData(register_addr);
611  }
612  else if(spi_irq_state == SEND_WRITE_DATA){
613 
614  /* Decrease the bytes to send */
615  byte_to_send--;
616 
617  /* Send data to write */
618  SPI_SendData(spi_buffer_ptr[buffer_idx]);
619 
620  /* Increase the buffer index */
621  buffer_idx++;
622  }
623  else {
624  /* Decrease the bytes to send */
625  byte_to_send--;
626 
627  /* Send dummy data */
628  SPI_SendData(0xFF);
629  }
630  }
631  else {
632  SPI_ITConfig(SPI_IT_TX , DISABLE);
633  if(spi_irq_state == SEND_WRITE_DATA)
634  spi_eot = SET;
635  }
636 
637  }
